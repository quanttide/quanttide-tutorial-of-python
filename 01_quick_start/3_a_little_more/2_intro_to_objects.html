class Person:
    def __init__(self, name, age):
        """
        类的构造器，我们在这里定义实例属性
        """
        # self.<属性> = 参数 or 经过计算
        self.name = name
        self.age = age
        
    def is_adult(self):
        """
        定义实例方法
        """
        if self.age>=18:
            return True
        else:
            return False
        
    def some_method(self, arg):
        return self.name + '_' + str(arg)
    
    def __str__(self):
        return "一个叫做" + self.name + "的人"
    
    def __repr__(self):
        return self.name
